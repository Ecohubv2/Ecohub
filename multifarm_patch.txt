
MULTI FARM + SELL ONLY FARMED ITEMS PATCH
-----------------------------------------

1) Add these variables near other Env variables:

Env.SelectedFarmTargets = {}
Env.CurrentFarmIndex = 1
Env.SellDelay = 5

-----------------------------------------

2) Add these helper functions under FarmData:

local function getCurrentFarmData()
    local name = Env.SelectedFarmTargets[Env.CurrentFarmIndex]
    if name then
        return FarmData[name]
    end
    return nil
end

local function nextFarm()
    Env.CurrentFarmIndex = Env.CurrentFarmIndex + 1
    if Env.CurrentFarmIndex > #Env.SelectedFarmTargets then
        Env.CurrentFarmIndex = 1
    end
end

-----------------------------------------

3) Replace the farm dropdown with this (multi select):

Tabs.Farm:AddDropdown("SelectFarmTarget", {
    Title = "Select Farm Target",
    Multi = true,
    Values = {"Strawberry","Corn","Chilli","Banana","Grape","Coconut","Pork","Flower","Wood","Grass"},
    Default = {},
    Callback = function(Value)
        Env.SelectedFarmTargets = Value
        Env.CurrentFarmIndex = 1
    end
})

-----------------------------------------

4) Add Sell Delay slider:

Tabs.Farm:AddSlider("SellDelay",{
    Title = "Sell Delay",
    Description = "Delay before selling",
    Default = 5,
    Min = 0,
    Max = 30,
    Rounding = 1,
    Callback = function(v)
        Env.SellDelay = v
    end
})

-----------------------------------------

5) Replace the selling logic with this so it sells ONLY selected farm items:

for _, farmName in pairs(Env.SelectedFarmTargets) do
    local data = FarmData[farmName]
    local amount = 0

    pcall(function()
        local body = Player.PlayerGui.Inventory.CanvasGroup.Main.Body
        for _, item in pairs(body:GetChildren()) do
            if item.Name == data.Item then
                local amt = item.Main.Amount.Text:match("^(%d+)")
                if amt then
                    amount = tonumber(amt)
                end
            end
        end
    end)

    if amount > 0 then
        repeat
            pcall(function()
                game:GetService("ReplicatedStorage")
                .Modules.NetworkFramework.NetworkEvent:FireServer(
                    "fire",
                    nil,
                    "Economy",
                    data.Item,
                    amount
                )
            end)

            task.wait(0.5)

            amount = 0
            pcall(function()
                local body = Player.PlayerGui.Inventory.CanvasGroup.Main.Body
                for _, item in pairs(body:GetChildren()) do
                    if item.Name == data.Item then
                        local amt = item.Main.Amount.Text:match("^(%d+)")
                        if amt then
                            amount = tonumber(amt)
                        end
                    end
                end
            end)

        until amount == 0
    end
end

-----------------------------------------

Result:
- Choose unlimited farm targets
- Farm them one by one
- Do NOT sell when first is full
- When ready to sell, it sells ONLY the selected farm items
- Sell delay adjustable
